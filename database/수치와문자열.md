```mssql
use BaseballData;

select 2021 - birthYear as koreanAge
from players
where deathYear is null and birthYear is not null and (2021 - birthYear) <= 80  -- 여기서는 koreanAge라는 alias를 사용할 수 없음
order by koreanAge asc;
```
