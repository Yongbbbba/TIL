# 월간 코드 챌린지 시즌1 

## 이진 변환 반복하기

그냥 단순한 구현문제 

### 코드

```c++
#include <bits/stdc++.h>

using namespace std;

#define endl '\n'
#define ll long long

string toBinary(int n)
{
	string r;
	while (n != 0) {
		r += (n % 2 == 0 ? "0" : "1");
		n /= 2;
	}
	return r;
}

vector<int> solution(string s)
{
	vector<int> answer;

	int convert = 0;
	int zeros = 0;

	while (true)
	{
		int len = s.length();

		int zero = 0;

		if (s == "1")
		{
			break;
		}

		for (auto& c : s)
		{
			if (c == '0')
			{
				zero++;
			}
		}

		zeros += zero;

		len -= zero;

		s = toBinary(len);
		convert++;
	}

	answer.push_back(convert);
	answer.push_back(zeros);

	return answer;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);

	string input = "1111111";

	vector<int> result = solution(input);
	for (auto& c : result)
	{
		cout << c << ' ';
	}

	return 0;
}
```
