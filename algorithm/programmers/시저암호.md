```python
def solution(s, n):
    lower_arr = [] # ascii
    upper_arr = [] # ascii 
    
    for i in range(26):
        lower_arr.append(i + ord('a'))
        upper_arr.append(i + ord('A'))
    
    answer = ''
    
    for c in s:
        if c.isupper() == True:
            idx = ord(c) - ord('A')
            idx = (idx + n) % 26
            answer += chr(upper_arr[idx])
        elif c.islower() == True:
            idx = ord(c) - ord('a')
            idx = (idx + n) % 26
            answer += chr(lower_arr[idx])
        else:
            answer += c
    
    return answer
```
