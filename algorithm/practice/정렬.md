# 선택, 버블 정렬 

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>

using namespace std;

int main()
{
	int nums[] = { 3,2,4,1,7,9,5,6 };

	// 선택 정렬
	// 최소값을 찾아서 i번째 원소에 집어넣기
	for (int i = 0; i < 8; i++)
	{
		swap(*min_element(nums + i, nums + 8), nums[i]);
	}

	for (int i = 0; i < 8; i++)
		cout << nums[i] << ' ';
	cout << endl;

	int nums2[] = { 3,2,4,1,7,9,5,6 };
	int n = 8;
	// 버블 정렬
	// 루프 한 번을 돌고 나면 i번째 자리에 가장 큰 수가 위치하게 된다.
	for (int i = n - 1; i > 0; i--)
	{
		for (int j = 0; j < i; j++)
		{
			if (nums2[j] > nums2[j + 1])
				swap(nums2[j], nums2[j + 1]);
		}
	}
	for (int i = 0; i < 8; i++)
		cout << nums2[i] << ' ';
	
}
```

# 병합 정렬

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>

using namespace std;

int arr[] = { 1,6,3,2,7,2,4,9,8,10 };
int temp[10];

void Merge(int s, int e)
{
	int m = (s + e) / 2;
	int l_idx = s;
	int r_idx = m;
	int idx = s;
	// 두 배열 합치기
	while (l_idx != m && r_idx != e)
	{
		if (arr[l_idx] < arr[r_idx])
			temp[idx++] = arr[l_idx++];
		else
			temp[idx++] = arr[r_idx++];
	}
	// 남은 놈 넣어주기
	if (l_idx != m)
	{
		while (l_idx != m)
			temp[idx++] = arr[l_idx++];
	}
	else
	{
		while (r_idx != e)
			temp[idx++] = arr[r_idx++];
	}
	for (int i = s; i < e; i++)
		arr[i] = temp[i];
}

void MergeSort(int s, int e)
{
	if (e == s+1)
		return;
	int m = (s + e) / 2;
	MergeSort(s, m);
	MergeSort(m, e);
	Merge(s, e);
}


int main()
{

	MergeSort(0, 10);

	for (int i = 0; i < 10; i++)
	{
		cout << arr[i] << ' ';
	}
	
}
 

```



## 삽입 정렬 

i번째 숫자와 왼쪽의 있는 수를 비교하면서 자신의 위치에 삽입하는 방법으로 구현

```c++
int main()
{
    vector<int> arr{ 7,5,9,0,3,1,6,2,4,8};
    
    for (int i = 1; i < arr.size(); i++)
    {
        for (int j = i; j > 0 ; j--)
        {
            if (arr[j] < arr[j - 1])
                swap(arr[j], arr[j - 1]);
            else
                break;
        } 
    }

    return 0;
}
```
