# boj 3986 좋은 단어

- 솔직히 풀이는 이해가 되는데 문제가 뭔소린지 이해가 안됨



## 코드

```c++
#include <bits/stdc++.h>

using namespace std;

#define endl '\n'
#define ll long long

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);

	int result = 0; 
	int n;
	cin >> n;

	while (n--)
	{
		string input; 
		cin >> input;
		stack<char> s;

		for (auto& c : input)
		{
			if (s.empty() || s.top() != c)
				s.push(c);
			else
				s.pop();
		}
		if (s.empty())
			result++;
	}
	cout << result;
	return 0;
}
```
