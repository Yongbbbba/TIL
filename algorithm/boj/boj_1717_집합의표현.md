# boj 1717 집합의 표현

- 간만에 유니온 파인드 복습하고 기본 문제를 풀었다. parent 배열을 기록할 때 부모노드를 기록하는 것이 아니라 루트 노드를 기록하였다. 



## 코드

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>

typedef long long ll;

using namespace std;
const int INF = 987654321;

int n, m;
//int parent[1000001];
vector<int> parent;

int findParent(int x)
{
    if (parent[x] == x)
        return  x;
    return parent[x] = findParent(parent[x]);
}

void unionParent(int a, int b)
{
    a = findParent(a);
    b = findParent(b);

    if (a < b)
        parent[b] = a;
    else
        parent[a] = b;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> m;
    parent.resize(n+1);
    for (int i = 1; i <= n; i++)
        parent[i] = i;

    for (int i = 0; i < m; i++)
    {
        int x, a, b;
        cin >> x >> a >> b;
        if (x == 0)
            unionParent(a, b);
        else
        {
            int aParent = findParent(a);
            int bParent = findParent(b);
            if (aParent == bParent)
                cout << "YES" << '\n';
            else
                cout << "NO" << '\n';
        }
    }
    return 0;
}
```
