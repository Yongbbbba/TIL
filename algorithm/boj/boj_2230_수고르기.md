# boj 2230 수 고르기

- 시간 제한이 빡빡하지 않아서 이중 for문 돌아도 통과는 한다.
- 정렬하고 이분 탐색으로 lower_bound로 구하면 매우 빠르게 통과
- 투포인터도 사용 가능 



## 이중 for 문

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>
#include <list>

typedef long long ll;
#define endl '\n'

using namespace std;
const int INF = 2100000000;

ll n, m;
vector<ll> arr;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> m;
    arr.resize(n);

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    ll result = INF;
    sort(arr.begin(), arr.end());
    
    for (int i = 0; i < n; i++)
    {
        for (int j = i; j < n; j++)
        {
            if (arr[j] - arr[i] >= m)
            {
                if (result > arr[j] - arr[i])
                    result = arr[j] - arr[i];
                break;
            }
        }
    }

    cout << result;
    
    return 0;
}
```



## 이분 탐색

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>
#include <list>

typedef long long ll;
#define endl '\n'

using namespace std;
const int INF = 2100000000;

ll n, m;
vector<ll> arr;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> m;
    arr.resize(n);

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    ll result = INF;
    sort(arr.begin(), arr.end());

    for (int i = 0; i < n; i++)
    {
        auto iter = lower_bound(arr.begin(), arr.end(), arr[i] + m);
        if (iter == arr.end())
            break;
        if (result > *iter - arr[i])
            result = *iter - arr[i];
    }

    cout << result;
    
    return 0;
}
```



## 투 포인터

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>
#include <list>

typedef long long ll;
#define endl '\n'

using namespace std;
const int INF = 2100000000;

ll n, m;
vector<ll> arr;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> m;
    arr.resize(n);

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    ll result = INF;
    sort(arr.begin(), arr.end());


    int st = 0;
    int en = 0;

    while (en < n && st < n)
    {
        if (arr[en] - arr[st] < m)
            en++;
        else
        {
            if (result > arr[en] - arr[st])
                result = arr[en] - arr[st];
            st++;
        }
    }

    cout << result;
    
    return 0;
}
```
