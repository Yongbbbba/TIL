# boj 2559 수열

- 단순 구현하면 O(n^2)으로 풀이할 수 있으나, 부분합을 이용하면 O(N)에 풀이 가능
- 아이디어는 바로 생각났는데 구현이 시간이 걸렸다.



## 코드

```c++
#include <bits/stdc++.h>

using namespace std;

#define endl '\n'
#define ll long long

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	int n, k;
	int result = -987654321;

	cin >> n >> k;
	vector<int> nums(n);

	for (int i = 0; i < n; i++)
	{
		int num;
		cin >> num;
		nums[i] = num;
	}

	int left = 0;
	int right = k - 1;
	int sum = 0;

	for (int i = 0; i < k; i++)
	{
		sum += nums[i];
	}
	result = sum;

	while (true)
	{
		sum -= nums[left];
		left++;
		right++;
		if (right == n)
		{
			break;
		}
		sum += nums[right];
		if (sum > result)
		{
			result = sum;
		}
	}

	cout << result;

	return 0;
}
```
