## 백트랙킹 풀이

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <fstream>

#define ll long long

using namespace std;

int T;
int result = 0;

void f(int num)
{
    if (num == 0)
    {
        result++;
        return;
    }
    if (num < 0)
        return;
    f(num - 1);
    f(num - 2);
    f(num - 3);
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> T;
    while (T--)
    {
        int num;
        cin >> num;
        f(num);
        cout << result << '\n';
        result = 0;
    }
    
}
```





## DP 풀이

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <fstream>

#define ll long long

using namespace std;

int T;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    // DP 테이블 만들기
    int dp[12];
    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 2;
    dp[3] = 4;
    dp[4] = 7;
    
    for (int i = 4; i <= 11; i++)
        dp[i] = dp[i - 3] + dp[i - 2] + dp[i - 1];
    
    cin >> T;
    while (T--)
    {
        int num;
        cin >> num;
        cout << dp[num] << '\n';
    }
    
}
```
