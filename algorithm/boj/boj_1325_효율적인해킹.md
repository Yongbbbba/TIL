# boj 1325 효율적인 해킹 

- 오늘은 문제 이해를 제대로 하지 않고 급하게 풀어서 구현에 미스가 좀 있었다. a가 b를 신뢰할 때 b를 해킹하면 a도 해킹하는 문제였는데, 양방향이라고 생각하고 풀어서 오답이 나왔었다.
- 풀이는 그냥 단순한게 BFS로 풀었음 



## 코드

```c++
int n, m;
vector<vector<int>> arr;


int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> m;
    arr.resize(n + 1);
    for (int i = 0; i < m; i++)
    {
        int a, b;
        cin >> a >> b;
        arr[b].push_back(a);
    }

    int maxV = 0;
    vector<int> result;

    for (int i = 1; i <= n; i++)
    {
        vector<int> vis(n+1);
        queue<int> q;
        q.push(i);
        vis[i] = 1;
        int cnt = 1;
        while (!q.empty())
        {
            int cur = q.front();
            q.pop();
            for (auto v : arr[cur])
            {
                if (vis[v] == 0)
                {
                    cnt++;
                    vis[v] = 1;
                    q.push(v);
                }
            }
        }
        if (cnt > maxV)
        {
            result.clear();
            result.push_back(i);
            maxV = cnt;
        }
        else if (cnt == maxV)
        {
            result.push_back(i);
        }
    }
    for (auto v : result)
    {
        cout << v << ' ';
    }

    return 0;
}
```
