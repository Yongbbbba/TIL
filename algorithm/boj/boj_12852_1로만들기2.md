# boj 12852 1로 만들기 2

- DP로 풀어봤는데 메모리도 많이 잡고 생각보다 구현이 오래걸렸다. 난이도가 높게 분류되어 있지 않은데, 괜히 어렵게 꼬아서 풀었나 싶은 생각이 든다.



## 코드

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <fstream>

#define ll long long

using namespace std;

int N;
int dp[1000001][2]; // 3은 3으로 나눈거, 2는 2로 나눈거, 1은 1을 뺀거

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> N;
    dp[2][0] = 1;
    dp[2][1] = 2;
    dp[3][0] = 1;
    dp[3][1] = 3;
    dp[4][0] = 2;  // dp[4-1], dp[4 / 2] 중 최솟값 + 1
    dp[4][1] = 2;

    for (int i = 5; i <= N; i++)
    {
        dp[i][0] = dp[i - 1][0] + 1;
        dp[i][1] = 1;
        if (i % 3 == 0)
        {
            if (dp[i][0] > dp[i / 3][0] + 1)
            {
                dp[i][0] = dp[i / 3][0] + 1;
                dp[i][1] = 3;
            }
        }
        if (i % 2 == 0)
        {
            if (dp[i][0] > dp[i / 2][0] + 1)
            {
                dp[i][0] = dp[i / 2][0] + 1;
                dp[i][1] = 2;
            }
        }
    }

    cout << dp[N][0] << '\n';
    while (N >= 1)
    {
        cout << N << ' ';
        if (dp[N][1] == 3)
            N /= 3;
        else if (dp[N][1] == 2)
            N /= 2;
        else
            N -= 1;
    }
}
```
