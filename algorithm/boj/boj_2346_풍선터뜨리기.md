# boj 2346 풍선 터뜨리기

- 처음이 더블 링크드 리스트를 구현해서 풀라고 하다가 실패.
- 리스트 STL 사용해서 풀려고 하다가 실패. 거의 써본적 없는 STL이라 사용법을 잘 모르겠더라.
- 위 방법으로 삽질하다가 시간 다 잡아먹고 그냥 배열에서 idx 예외처리해줘서 풀었다..
- if문이 남발하는 지저분한 코드가 되어버림



## 코드

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>
#include <list>

typedef long long ll;
#define endl '\n'

using namespace std;
const int INF = 987654321;

int n;
vector<int> arr;

int move(int num, int idx)
{
    // num의 방향
    // arr의 값이 0인지
    // 1~n의 범위를 idx가 벗어낫는지
    while (num != 0)
    {
        if (num > 0)
        {
            idx++;
            if (idx == n+1)
            {
                idx = 1;
                if (arr[idx] == 0)
                    continue;
                num--;
            }
            else
            {
                if (arr[idx] == 0)
                    continue;
                num--;
            }
        }
        else
        {
            idx--;
            if (idx == 0)
            {
                idx = n;
                if (arr[idx] == 0)
                    continue;
                num++;
            }
            else
            {
                if (arr[idx] == 0)
                    continue;
                num++;
            }
        }
    }

    return idx;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n;
    arr.resize(n+1);
    for (int i = 1; i <= n; i++)
        cin >> arr[i];

    int idx = 1;

    for (int i = 1; i <= n; i++)
    {
        cout << idx << ' ';
        if (i == n)
            break;
        int num = arr[idx];
        arr[idx] = 0;
        idx = move(num, idx);
    }
    
    return 0;
}
```
