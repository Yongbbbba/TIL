# BOj_1463_1로 만들기

- BFS 배우기 전에는 5번인가 틀리고 포기했었다. BFS 배우고나서 푸니까 매우 쉬운 문제였다는 것을 알게 되었다.



## 코드

```c++
#include <bits/stdc++.h>
#define x 1000001
using namespace  std;

int main()
{
    int N;
    cin >> N;
    queue<int> q;
    int result = 0;
    int arr[x];
    fill(arr, arr+x, -1);
    q.push(N);
    arr[N] = 0;
    
    while (!q.empty())
    {
        int n = q.front();
        q.pop();
        
        // 종료 조건
        if (n==1)
        {
            result = arr[n];
            break; 
        }

        for (auto c : {n%3 == 0? n/3 : -1 , n%2 == 0? n/2 : -1, n-1})
        {
            if (c!= -1 && arr[c] < 0)
            {
                q.push(c);
                arr[c] = arr[n] + 1;
            }
        }
    }
    cout << result;
    return 0;
}
```
