# boj 2947 나무 조각

- 실버5 치고는 너무 쉬웠다.. 그냥 단순 구현 문제



## 코드 

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <set>
#include <sstream>
#include <map>

typedef long long ll;
#define endl '\n'

using namespace std;
const int INF = 987654321;

int arr[5];
void printArr()
{
    for (int i = 0; i < 5; i++)
    {
        cout << arr[i] << ' ';
    }
    cout << endl;
}

bool isAnswer()
{
    for (int i = 1; i <= 5; i++)
    {
        if (arr[i - 1] != i)
            return false;
    }
    return true;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    for (int i = 0; i < 5; i++)
        cin >> arr[i];

    while (true)
    {
        if (arr[0] > arr[1])
        {
            swap(arr[0], arr[1]);
            printArr();
        }
        if (arr[1] > arr[2])
        {
            swap(arr[1], arr[2]);
            printArr();
        }
        if (arr[2] > arr[3])
        {
            swap(arr[2], arr[3]);
            printArr();
        } 
        if (arr[3] > arr[4])
        {
            swap(arr[3], arr[4]);
            printArr();
        }
        if (isAnswer())
            break;
    }

    return 0;
}
```
