# boj 11724 연결 요소의 개수

- DFS를 이용
- 방향성 없는 그래프의 경우 arr[u] = v, arr[v] = u를 입력받을 때 양쪽 노드를 같이 해줘야하는데 한군데만 해서 처음에 오답.

## 코드

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>
#include <fstream>
#include <set>
#include <sstream>
#include <unordered_set>

typedef long long ll;

using namespace std;

int N, M;
int result;
int vis[1001];
vector<vector<int>> arr(1001, vector<int>());
    
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> N >> M;
    for (int i = 0; i < M; i++)
    {
        int u, v;
        cin >> u >> v;
        arr[u].push_back(v);
        arr[v].push_back(u);
    }
    stack<int> st;
    for (int i = 1; i <= N; i++)
    {
        if (vis[i] == 0)
        {
            for (auto node : arr[i])
            {
                st.push(node);
                vis[node] = 1;
            }
            while (!st.empty())
            {
                int cur = st.top();
                st.pop();
                for (auto node : arr[cur])
                {
                    if (vis[node] == 0)
                    {
                        st.push(node);
                        vis[node] = 1;
                    }
                }
            }
            result++;
        }
    }
    
    cout << result;
    
    return 0;
}
```
