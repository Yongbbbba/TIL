# BOJ_10866_덱



- 이상하게 머리가 꼬여버린건지 내가 그냥 원래 멍청한건지 도저히 구현이 안되더라... 
- 원래 `deque` 안 쓰고 직접 구현해서 할라다가 포기했다. 다른 코드 살펴보고 나서야 구현할 수 있었다.
- 배열을 이용한 deque 구현의 핵심은 앞뒤로 원소가 들어가기 때문에 배열의 중간쯤부터 시작을 해야한다는 것이다.
- 메모리 공간 낭비를 줄이기 위해서 원형으로 덱을 구성할 수도 있다.



## 코드 1 : deque 사용

```c++
#include <iostream>
#include <deque>

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    
    int N;
    cin >> N;
    deque<int> dq;
    string order;
    int num;
    for (int i=0; i<N; i++)
    {
        cin >> order;
        if (order == "push_front")
        {
            cin >> num;
            dq.push_front(num);
        }
        else if (order == "push_back")
        {
            cin >> num;
            dq.push_back(num);
        }
        else if (order == "pop_front")
        {
            if (dq.size() == 0) cout << -1 << '\n';
            else
            {
                cout << dq.front() << '\n';
                dq.pop_front();
            }
        }
        else if (order == "pop_back")
        {
            if (dq.size() == 0) cout << -1 << '\n';
            else
            {
                cout << dq.back() << '\n';
                dq.pop_back();
            }
        }
        else if (order == "size") cout << dq.size() << '\n';
        else if (order == "empty")
        {
            if (dq.empty()) cout << 1 << '\n';
            else cout << 0 << '\n';
        }
        else if (order == "front") 
        {
            if (!dq.empty())cout << dq.front() << '\n';
            else cout << -1 << '\n';
        }
        else if (order == "back") 
        {
            if (!dq.empty())cout << dq.back() << '\n';
            else cout << -1 << '\n';
        }


    }

    return 0;
    
}
```





## 코드 2 : 직접 구현해서 사용

```c++
#include <iostream>

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    
    int N;
    cin >> N;
    int MX = 10000;
    int dq[2*MX +1];
    string order;
    int size = 0;
    int head = MX;
    int tail = MX;
    int num;
    for (int i=0; i<N; i++)
    {
        cin >> order;
        if (order == "push_front")
        {
            cin >> num;
            dq[--head] = num;
            size++;
        }
        else if (order == "push_back")
        {
            cin >> num;
            dq[tail++] = num;
            size++;
        }
        else if (order == "pop_front")
        {
            if (size == 0) cout << -1 << '\n';
            else 
            {
                cout << dq[head++] << '\n';
                size--;
            }
        }
        else if (order == "pop_back")
        {
            if (size == 0) cout << -1 << '\n';
            else 
            {
                cout << dq[tail-1] << '\n';
                tail--;
                size--;
            }
        }
        else if (order == "size") cout << size << '\n';
        else if (order == "empty")
        {
            if (size==0) cout << 1 << '\n';
            else cout << 0 << '\n';
        }
        else if (order == "front") 
        {
            if (size != 0)cout << dq[head] << '\n';
            else cout << -1 << '\n';

        {

            else cout << -1 << '\n';
        }


    }

    return 0;
    
}
```

## 1년 뒤 풀이 : 2022-09-14

- 단순 구현인데 빼먹고 구현한 부분도 있었고, pop, push할 때 head, tail을 바꿔주는 부분을 잘못 생각해서 틀렸다.

### 코드 

```c++
#include <bits/stdc++.h>

using namespace std;

#define endl '\n'
#define ll long long

const int MX = 10000005;
int dat[MX * 2 + 1];
int head = MX, tail = MX;

void push_front(int x)
{
	dat[--head] = x;
}

void push_back(int x)
{
	dat[tail++] = x;
}

int pop_front()
{
	if (head == tail)
	{
		return -1;
	}
	int result = dat[head];
	head++;
	return result;
}

int pop_back()
{
	if (head == tail)
	{
		return -1;
	}
	int result = dat[tail-1];
	tail--;
	return result;
}

int front()
{
	if (head == tail)
	{
		return -1;
	}
	return dat[head];
}

int back()
{
	if (head == tail)
	{
		return -1;
	}
	return dat[tail-1];
}

int size()
{
	return tail - head;
}

int empty()
{
	if (head == tail)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);

	int n; 
	cin >> n;
	while (n--)
	{
		string oper;
		cin >> oper; 

		if (oper == "push_back")
		{
			int num;
			cin >> num;
			push_back(num);
		}
		else if (oper == "push_front")
		{
			int num;
			cin >> num;
			push_front(num);
		}
		else if (oper == "front")
		{
			cout << front() << endl;
		}
		else if (oper == "back")
		{
			cout << back() << endl;
		}
		else if (oper == "size")
		{
			cout << size() << endl;
		}
		else if (oper == "empty")
		{
			cout << empty() << endl;
		}
		else if (oper == "pop_back")
		{
			cout << pop_back() << endl;
		}
		else if (oper == "pop_front")
		{
			cout << pop_front() << endl;
		}
	}

	return 0;
}
```
