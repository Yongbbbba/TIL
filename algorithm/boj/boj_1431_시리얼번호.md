# boj 1431 시리얼 번호

- sort에 비교 함수를 넣는 것을 활용하는 문제. cmp를 이렇게까지 활용해본 것은 처음인데 신기하다.

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <cctype>
#include <stack>
#include <queue>
#include <array>

using namespace std;
int N;

bool cmp(string& a, string& b)
{
	int sum_a = 0, sum_b = 0;
	for (auto c : a)
	{
		if (isdigit(c))
			sum_a += static_cast<int>(c - '0');
	}
	for (auto c : b)
	{
		if (isdigit(c))
			sum_b += static_cast<int>(c - '0');
	}
	// 길이가 짧은 것이 먼저 옴
	// 만약 같으면 숫자의 합이 작은 것이 먼저 옴
	// 합도 같으면 사전순으로 비교
	if (a.length() != b.length())
		return a.length() < b.length();
	else if (sum_a != sum_b)
		return sum_a < sum_b;
	else
		return a < b;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	cin >> N;
	vector<string> input(N);

	for (int i = 0; i < N; i++)
	{
		cin >> input[i];
	}

	sort(input.begin(), input.end(), cmp);
	for (auto& s : input)
		cout << s << '\n';
}
```
